const express = require('express'); const app = express(); app.use(express.json()); // Dummy data const users = [ { voterId: "123", name: "Arun", constituency: "Salem" }, { voterId: "456", name: "Kumar", constituency: "Chennai" } ]; const candidates = { Salem: ["Candidate A", "Candidate B"], Chennai: ["Candidate X", "Candidate Y"] }; let votes = []; // Serve frontend app.get('/', (req, res) => { res.send(`

Digital Voting System

`); }); // Login API app.post('/login', (req, res) => { const { voterId } = req.body; const user = users.find(u => u.voterId === voterId); if (!user) return res.status(404).send("Invalid ID"); res.json(user); }); // Get candidates app.get('/candidates/:constituency', (req, res) => { res.json(candidates[req.params.constituency] || []); }); // Vote API app.post('/vote', (req, res) => { const { voterId, candidate } = req.body; const already = votes.find(v => v.voterId === voterId); if (already) return res.status(400).send("Already voted"); votes.push({ voterId, candidate }); res.send("Vote recorded successfully"); }); app.listen(3000, () => console.log("Server running on port 3000"));
